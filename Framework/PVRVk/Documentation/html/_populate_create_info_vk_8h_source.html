83 inline void populateShaderInfo(
84  const VkShaderModule& shader, VkShaderStageFlags vkShaderStage,
85  VkSpecializationInfo& specializationInfo, char* specializationInfoData,
86  const ShaderConstantInfo* shaderConsts, uint32_t shaderConstCount,
87  VkSpecializationMapEntry* mapEntries,
88  VkPipelineShaderStageCreateInfo& outShader, const char* entryPoint)
89 {
90  // caculate the number of size in bytes required.
91  uint32_t specicalizedataSize = 0;
92  for (uint32_t i = 0; i < shaderConstCount; ++i)
93  {
94  specicalizedataSize += shaderConsts[i].sizeInBytes;
95  }
96 
97  if (specicalizedataSize)
98  {
99  debug_assertion(specicalizedataSize < FrameworkCaps::MaxSpecialisationInfoDataSize,
100  "Specialised Data out of range.");
101  uint32_t dataOffset = 0;
102  for (uint32_t i = 0; i < shaderConstCount; ++i)
103  {
104  memcpy(&specializationInfoData[dataOffset], shaderConsts[i].data, shaderConsts[i].sizeInBytes);
105  mapEntries[i] = VkSpecializationMapEntry{ shaderConsts[i].constantId, dataOffset, shaderConsts[i].sizeInBytes };
106  dataOffset += shaderConsts[i].sizeInBytes;
107  }
108  specializationInfo.mapEntryCount = shaderConstCount;
109  specializationInfo.pMapEntries = mapEntries;
110  specializationInfo.dataSize = specicalizedataSize;
111  specializationInfo.pData = specializationInfoData;
112  }
113 
114  outShader.sType = VkStructureType::e_PIPELINE_SHADER_STAGE_CREATE_INFO;
115  outShader.pNext = nullptr;
116  outShader.flags = 0;
117  outShader.pSpecializationInfo = (specicalizedataSize ? &specializationInfo : nullptr);
118  outShader.stage = vkShaderStage;
119  outShader.module = shader;
120  outShader.pName = entryPoint;
