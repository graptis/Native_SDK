287  bool getMemoryTypeIndex(
288  const VkPhysicalDeviceMemoryProperties& deviceMemProps,
289  uint32_t typeBits, VkMemoryPropertyFlags properties,
290  uint32_t& outTypeIndex)
291  {
292  for (;;)
293  {
294  uint32_t typeBitsTmp = typeBits;
295  for (uint32_t i = 0; i < 32; ++i)
296  {
297  if ((typeBitsTmp & 1) == 1)
298  {
299  if (VkMemoryPropertyFlags(deviceMemProps.memoryTypes[i].propertyFlags & properties) == properties)
300  {
301  outTypeIndex = i;
302  return true;
303  }
304  }
305  typeBitsTmp >>= 1;
306  }
307  if ((properties & VkMemoryPropertyFlags::e_LAZILY_ALLOCATED_BIT) != 0)
308  {
309  properties = VkMemoryPropertyFlags(properties & ~VkMemoryPropertyFlags::e_LAZILY_ALLOCATED_BIT);
310  continue;
311  }
312  if ((properties & VkMemoryPropertyFlags::e_DEVICE_LOCAL_BIT) != 0)
313  {
314  properties = VkMemoryPropertyFlags(properties & ~VkMemoryPropertyFlags::e_DEVICE_LOCAL_BIT);
315  continue;
316  }
317  if ((properties & VkMemoryPropertyFlags::e_LAZILY_ALLOCATED_BIT) != 0)
318  {
319  properties = VkMemoryPropertyFlags(properties & ~VkMemoryPropertyFlags::e_LAZILY_ALLOCATED_BIT);
320  continue;
321  }
