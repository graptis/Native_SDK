291  if (size > width * height && w >= width && h >= height)

294  left = new Area;
295  right = new Area;
296  left->x = x;
297  left->y = y;
298

300  // Splits vertically if larger free distance across the texture.
301  if ((w - width) > (h - height))
302  {

304  left->h = h;
305 
306  right->x = x + width;
307  right->y = y;
308  right->w = w - width;
309  right->h = h;
310  }
311  // Splits horizontally if larger or equal free distance downwards.
312  else

969 inline void create3dPlaneMesh(uint32_t width, uint32_t length, bool vertexAttribTex, bool vertexAttribNormal,
970  assets::Mesh& outMesh)
971 {
972  const float halfWidth = width * .5f;
973  const float halfLength = length * .5f;
974 
975  glm::vec3 normal[4] =
976  {
977  glm::vec3(0.0f, 1.0f, 0.0f),
978  glm::vec3(0.0f, 1.0f, 0.0f),
979  glm::vec3(0.0f, 1.0f, 0.0f),
980  glm::vec3(0.0f, 1.0f, 0.0f)
981  };
982 
983  glm::vec2 texCoord[4] =
984  {
985  glm::vec2(0.0f, 1.0f),
986  glm::vec2(0.0f, 0.0f),
987  glm::vec2(1.0f, 0.0f),
988  glm::vec2(1.0f, 1.0f),
989  };
990 
991  glm::vec3 pos[4] =
992  {
993  glm::vec3(-halfWidth, 0.0f, -halfLength),
994  glm::vec3(-halfWidth, 0.0f, halfLength),
995  glm::vec3(halfWidth, 0.0f, halfLength),
996  glm::vec3(halfWidth, 0.0f, -halfLength)
997  };
998 
999  uint32_t indexData[] =
1000  {
1001  0, 1, 2,
1002  0, 2, 3
1003  };
1004 
1005  float vertData[32];
1006  uint32_t offset = 0;
1007 
1008  for (uint32_t i = 0; i < 4; ++i)
1009  {
1010  memcpy(&vertData[offset], &pos[i], sizeof(pos[i]));
1011  offset += 3;
1012  if (vertexAttribNormal)
1013  {
1014  memcpy(&vertData[offset], &normal[i], sizeof(normal[i]));
1015  offset += 3;
1016  }
1017  if (vertexAttribTex)
1018  {
1019  memcpy(&vertData[offset], &texCoord[i], sizeof(texCoord[i]));
1020  offset += 2;
1021  }
1022  }
1023 
1024  uint32_t stride =
1025  sizeof(glm::vec3) +
1026  (vertexAttribNormal ? sizeof(glm::vec3) : 0) +
1027  (vertexAttribTex ? sizeof(glm::vec2) : 0);
1028 
1029  outMesh.addData((const uint8_t*)vertData, sizeof(vertData), stride, 0);
1030  outMesh.addFaces((const uint8_t*)indexData, sizeof(indexData), IndexType::IndexType32Bit);
1031  offset = 0;
1032  outMesh.addVertexAttribute("POSITION", DataType::Float32, 3, offset, 0);
1033  offset += sizeof(float) * 3;
1034  if (vertexAttribNormal)
1035  {
1036  outMesh.addVertexAttribute("NORMAL", DataType::Float32, 3, offset, 0);
1037  offset += sizeof(float) * 2;
1038  }
1039  if (vertexAttribTex)
1040  {
1041  outMesh.addVertexAttribute("UV0", DataType::Float32, 2, offset, 0);
1042  }
1043  outMesh.setPrimitiveType(PrimitiveTopology::TriangleList);
1044  outMesh.setStride(0, stride);
1045  outMesh.setNumFaces(ARRAY_SIZE(indexData) / 3);
1046  outMesh.setNumVertices(ARRAY_SIZE(pos));
1047 }
1048 }
