283  void calcSizeAndOffset(const uint32_t& offset)

568  _prototype(entry), _level(level), _mappedMemory(nullptr)
569  {
570  _indices[0] = elementArrayIndex;

577  void init(uint32_t dynamicSlice = 0)
578  {
579  // Get the offset INSIDE CURRENT LEVEL.
580  _offset = _prototype.getArrayElementOffset(_indices[0]);
581  const StructuredMemoryEntry* parent = _prototype.getParent();
582  size_t level = 1; // How many levels up we have gone
583  uint64_t dynamicSliceSize = 0;
584  uint32_t mappedDynamicSlice = 0;
585  while (parent) // Until we reach the root element
586  {
587  debug_assertion(parent->getNumArrayElements() > _indices[level], "StructuredBufferViewElement: Attempted out-of-bounds access in getOffset");
588  _offset += parent->getArrayElementOffset(_indices[level++]);
589  dynamicSliceSize = parent->getSize();
590  mappedDynamicSlice = parent->getMappedDynamicSlice();
591  // store the mapped memory so we only have to do this lookup once rather than each time setValue is called
592  _mappedMemory = parent->getMappedMemory();
593  parent = parent->getParent();
594  }
595 
596  // at this point dynamicSliceSize matches the root size
597  debug_assertion(dynamicSlice >= mappedDynamicSlice, "StructuredBufferViewElement: Mapped dynamic slice must be greater than or equal to the current dynamic slice");
598  uint32_t sliceOffset = (dynamicSlice * static_cast<uint32_t>(dynamicSliceSize)) - (mappedDynamicSlice * static_cast<uint32_t>(dynamicSliceSize));
599  _offset += sliceOffset;
600  }
