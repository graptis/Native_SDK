177 inline void MemorySuballocation_::recycle()
178 {
179  if (_suballocator.isNull())
180  {
181  return;
182  }
183  auto& freemem = _suballocator->_freeMemory;
184  if (freemem.empty()) // Allocator completely full. Insert this as the only free chunk.
185  {
186  freemem.push_back(MemorySuballocator_::AllocationPair(offset(), size()));
187  }
188  else // Allocator not completely full
189  {
190  for (size_t i = 0; i < freemem.size(); ++i) //Try to find where the chunk must be placed, handling it being contiguous to other chunks.
191  {
192  auto&& current_chunk = freemem[i];
193  if (_offset < current_chunk.start_offset) // This is the place to put it: it must be placed before the chunk under consideration.
194  {
195  if (_offset + _size == current_chunk.start_offset) // It is actually contiguous to its next chunk - we'll use this!
196  {
197  current_chunk.start_offset = _offset;
198  // Now we must additionally check that we did not introduce extra contiguousness: the new chunk exactly filling a gap with the previous chunk
199  if (i > 0 && current_chunk.start_offset == freemem[i - 1].end())
200  {
201  // It does - so merge them!
202  current_chunk.start_offset = freemem[i - 1].start_offset;
203  freemem.erase(freemem.begin() + i - 1);
204  }
205  }
206  else // ... No contiguousness with the next chunk
207  {
208  if (i > 0 && offset() + size() == freemem[i - 1].end()) //Check for contiguousness with the previous chunk
209  {
210  // Contiguous with the previous chunk:
211  freemem[i - 1].size += size();
212  }
213  else // No contiguousness...
214  {
215  freemem.insert(freemem.begin(), MemorySuballocator_::AllocationPair(offset(), size()));
216  }
217  }
218  this->_suballocator.reset();
219  this->_size = 0;
220  this->_offset = 0;
221  return;
222  }
223  }
224 
225  if (_offset + _size == freemem.back().end())
226  {
227  freemem.back().size += _size;
228  }
229  else
230  {
231  freemem.push_back(MemorySuballocator_::AllocationPair(offset(), size()));
232  }
233  }
234  this->_suballocator.reset();
235  this->_size = 0;
236  this->_offset = 0;
237 }

239 inline MemorySuballocation MemorySuballocator_::suballocate(VkDeviceSize size)
240 {
241  MemorySuballocation retval;
242  auto aligned_size = align(size, _alignment);
243  for (size_t i = 0; i < _freeMemory.size(); ++i)
244  {
245  auto&& chunk = _freeMemory[i];
246  if (chunk.size >= aligned_size)
247  {
248  if (aligned_size)
249  { retval.construct(getReference(), aligned_size, chunk.start_offset); }
250 
251  chunk.start_offset += aligned_size;
252  chunk.size -= aligned_size;
253  if (chunk.size == 0)
254  {
255  _freeMemory.erase(_freeMemory.begin() + i);
256  }
257  return retval; //SUCCESSFUL RETURN HERE
258  }
259  }
260  Log("MemorySuballocation: Could not find enough contiguous space for allocation of size %ull (aligned from %ull)", aligned_size, size);
