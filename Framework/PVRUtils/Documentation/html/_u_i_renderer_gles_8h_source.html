230  const GLuint& getFontIbo()
231  {
232  if (!_fontIboCreated)
233  {
234  // create the FontIBO
235  std::vector<uint16_t> fontFaces;
236  fontFaces.resize(impl::Font_::FontElement);
237 
238  for (uint32_t i = 0; i < impl::Font_::MaxRenderableLetters; ++i)
239  {
240  fontFaces[i * 6] = 0 + i * 4;
241  fontFaces[i * 6 + 1] = 3 + i * 4;
242  fontFaces[i * 6 + 2] = 1 + i * 4;
243 
244  fontFaces[i * 6 + 3] = 3 + i * 4;
245  fontFaces[i * 6 + 4] = 0 + i * 4;
246  fontFaces[i * 6 + 5] = 2 + i * 4;
247  }
248  GLint binding;
249  gl::GetIntegerv(GL_ELEMENT_ARRAY_BUFFER_BINDING, &binding);
250 
251  gl::GenBuffers(1, &_fontIbo);
252  gl::BindBuffer(GL_ELEMENT_ARRAY_BUFFER, _fontIbo);
253  gl::BufferData(GL_ELEMENT_ARRAY_BUFFER, static_cast<uint32_t>(sizeof(fontFaces[0]) * fontFaces.size()), &fontFaces[0], GL_STATIC_DRAW);
254  gl::BindBuffer(GL_ELEMENT_ARRAY_BUFFER, binding);
255 
256  _uiStateTracker.ibo = _fontIbo;
257  _fontIboCreated = true;
258  }
259  return _fontIbo;
