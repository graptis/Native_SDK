116 inline bool isExtensionSupported(const cl::Platform& platform, const char* extension)
117 {
118  // Extension names should not have spaces.
119  const char* where = strchr(extension, ' ');
120 
121  if (where || *extension == '\0')
122  {
123  return 0;
124  }
125 
126  std::string extensions;
127  platform.getInfo(CL_PLATFORM_EXTENSIONS, &extensions);
128 
129  const char* start;
130  const char* terminator;
131 
132  start = extensions.c_str();
133  for (;;)
134  {
135  where = strstr(start, extension);
136  if (!where)
137  {
138  break;
139  }
140  terminator = where + strlen(extension);
141  if (where == start || *(where - 1) == ' ')
142  {
143  if (*terminator == ' ' || *terminator == '\0')
144  {
145  return true;
146  }
147  }
148  start = terminator;
149  }
150  return false;
151 }

153 inline bool createOpenCLContext(
154  cl::Platform& outPlatform, cl::Device& outDevice, cl::Context& outContext, cl::CommandQueue& outQueue,
155  cl_command_queue_properties queue_properties = 0, cl_device_type device_type = CL_DEVICE_TYPE_ALL,
156  const char* const platformName = NULL, cl_int* err = 0)
157 {
158  bool contextCreated = false;
159  cl_int errstr;
160  cl_int& errcode = err ? *err : errstr;
161 
162  /*
163  * Query the available OpenCL platforms.
164  */
165  std::vector<cl::Platform> platforms;
166  errcode = cl::Platform::get(&platforms);
167  if (errcode != CL_SUCCESS)
168  {
169  Log(LogLevel::Error, "createOpenCLContext : Failed to query platform IDs with code %s.", getOpenCLError(errcode));
170  return false;
171  }
172 
173  if (platforms.size() == 0)
174  {
175  Log(LogLevel::Error, "createOpenCLContext : No OpenCL capable platform found with code %s.", getOpenCLError(errcode));
176  return false;
177  }
178 
179  /*
180  * Iterate over all of the available platforms until one is found that matches the requirements.
181  */
182  for (cl_uint i = 0; i < platforms.size(); i++)
183  {
184  /*
185  * Check whether the platform matches the requested one.
186  */
187  std::string platName;
188  size_t platformNameLength = 0;
189  errcode = platforms[i].getInfo(CL_PLATFORM_NAME, &platName);
190  if (errcode != CL_SUCCESS)
191  {
192  Log(LogLevel::Error, "createOpenCLContext: Error: Failed to query platform name with code %s.", getOpenCLError(errcode));
193  continue ;
194  }
195  if (platformName)
196  {
197  bool bPlatformNameMatches = false;
198  if (strncmp(platName.c_str(), platformName, platformNameLength) == 0)
199  { bPlatformNameMatches = true; }
200 
201  // Continue with the next platform if the current one is unsuitable.
202  if (!bPlatformNameMatches)
203  { continue; }
204  }
205 
206  /*
207  * Query for the first available device that matches the requirements.
208  */
209  std::vector<cl::Device> devices;
210  errcode = platforms[i].getDevices(device_type, &devices);
211  if (errcode != CL_SUCCESS && errcode != CL_DEVICE_NOT_FOUND)
212  {
213  Log(LogLevel::Error, "createOpenCLContext: Failed to query OpenCL devices with code %s.", getOpenCLError(errcode));
214  return false;
215  }
216 
217  if (devices.size() == 0)
218  {
219  // This platform does not have a suitable device, continue with the next platform.
220  continue;
221  }
222 
223  std::string platform_extensions;
224  platforms[i].getInfo(CL_PLATFORM_EXTENSIONS, &platform_extensions);
225  std::string device_extensions;
226  devices[0].getInfo(CL_DEVICE_EXTENSIONS, &device_extensions);
227 
228  cl_platform_id platId = platforms[i].Wrapper<cl_platform_id>::get();
229  cl_context_properties contextProperties[] =
230  {
231  CL_CONTEXT_PLATFORM, (cl_context_properties)platId,
232  0, 0,
233  0, 0,
234  0, 0,
235  0,
236  };
237 
238  cl::Context context = cl::Context(devices, contextProperties, nullptr, nullptr, &errcode);
239  if (errcode != CL_SUCCESS)
240  {
241  Log(LogLevel::Error, "createOpenCLContext: Failed to create context with code %s.", getOpenCLError(errcode));
242  return false;
243  }
244 
245 #ifdef __APPLE__
246  cl_queue_properties_APPLE props[] = { 0 };
247 
248  cl_command_queue q =
249  clCreateCommandQueueWithPropertiesAPPLE(context.get(), devices[0].get(), props, &errcode);
250 #else
251  cl_command_queue q;
252  {
253  cl_command_queue_properties props = { NULL };
254  q = clCreateCommandQueue(context.get(), devices[0].get(), props, &errcode);
255  }
256 #endif
257  cl::CommandQueue queue(q);
258  if (errcode != CL_SUCCESS || q == NULL)
259  {
260  Log(LogLevel::Error, "createOpenCLContext: Failed to create command queue with code %s.", getOpenCLError(errcode));
261  return false;
262  }
263 
264  Log(LogLevel::Information, "createOpenCLContext: Created context on platform %s.", platName.c_str());
265  outPlatform = platforms[i];
266  outContext = context;
267  outDevice = devices[0];
268  outQueue = queue;
269  contextCreated = true;
270  break;
271  }
272 
273  if (!contextCreated)
274  {
275  Log(LogLevel::Error, "createOpenCLContext: No suitable OpenCL context(platform, context, device and queue) found");
276  return false;
277  }
278 
279  return true;
280 }

283 {
284  cl_int errcode;
285 
286  // Append define's here if there are any
287  std::string pszKernelString;
288 
289  if (!kernelSource.isopen() && !kernelSource.open())
290  {
291  Log(LogLevel::Error, "loadShader: Could not open the shaderSource stream.");
292  return cl::Program();
293  }
294  std::string shaderSrc;
295  if (!kernelSource.readIntoString(shaderSrc))
296  {
297  Log(LogLevel::Error, "Error: Failed to load the OpenCL shader");
298  return cl::Program();
299  };
300 
301  std::string sourceDataStr;
302  // insert the defines
303  for (uint32_t i = 0; i < defineCount; ++i)
304  {
305  sourceDataStr.append("#define ");
306  sourceDataStr.append(defines[i]);
307  sourceDataStr.append("\n");
308  }
309  sourceDataStr.append("\n");
310  sourceDataStr.append(shaderSrc);
311 
312  cl::Program program(ctx, sourceDataStr, false, &errcode);
313  if (program.get() == NULL || errcode != CL_SUCCESS)
314  {
315  Log(LogLevel::Error, "Error: Failed to create OpenCL program with code %s", getOpenCLError(errcode));
316  return cl::Program();
317  }
318  errcode = program.build();
319  if (errcode != CL_SUCCESS)
320  {
321  std::string errlog;
322  errcode = program.getBuildInfo<std::string>(device, CL_PROGRAM_BUILD_LOG, &errlog);
323  if (errcode != CL_SUCCESS)
324  {
325  Log(LogLevel::Error, "Error: Failed to get build log with code %s", getOpenCLError(errcode));
326  return cl::Program();
327  }
328  Log(LogLevel::Error, errlog.c_str());
329  return cl::Program();
330  }
331  return program;
