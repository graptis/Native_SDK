296  // INTERNAL NOTE: CALL THIS ** AFTER ** CALC BASE ALIGNMENT, TO FIX THE _offset AND _size MEMBERS
297  void calcSizeAndOffset(const uint32_t& offset)
298  {
299  _offset = align(offset, _baseAlignment);
300  _offset = align(_offset, _minDynamicAlignment);
301  if (isStructure())

305  {
306  child.calcSizeAndOffset(tmp_offset);

308  }
309  // STD140: Structures are padded to their alignment so that a[n] ==> sizeof(a[0]) == sizeof[a] / n

632  void init(uint32_t dynamicSlice = 0)
633  {
634  // Get the offset INSIDE CURRENT LEVEL.
635  _offset = _prototype.getArrayElementOffset(_indices[0]);
636  const StructuredMemoryEntry* parent = _prototype.getParent();
637  size_t level = 1; // How many levels up we have gone
638  uint64_t dynamicSliceSize = 0;
639  uint32_t mappedDynamicSlice = 0;
640  while (parent) // Until we reach the root element
641  {
642  debug_assertion(parent->getNumArrayElements() > _indices[level], "StructuredBufferViewElement: Attempted out-of-bounds access in getOffset");
643  _offset += parent->getArrayElementOffset(_indices[level++]);
644  dynamicSliceSize = parent->getSize();
645  mappedDynamicSlice = parent->getMappedDynamicSlice();
646  // store the mapped memory so we only have to do this lookup once rather than each time setValue is called
647  _mappedMemory = parent->getMappedMemory();
648  parent = parent->getParent();
649  }
650

939  {

943  }

958  memcpy((char*)getMappedMemory() + (size_t)myoff, &tmp[0][0], (size_t)myvaluesize);
959  }
960  }

962  {

964  uint64_t myvaluesize = value.dataSize();

966  }
