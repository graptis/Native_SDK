292  // INTERNAL NOTE: CALL THIS ** AFTER ** CALC BASE ALIGNMENT, TO FIX THE _offset AND _size MEMBERS
293  void calcSizeAndOffset(const uint32_t& offset)
294  {
295  _offset = align(offset, _baseAlignment);
296  _offset = align(_offset, _minDynamicAlignment);
297  if (isStructure())

299  uint32_t tmp_offset = 0;
300  for (auto& child : _childEntries)
301  {
302  child.calcSizeAndOffset(tmp_offset);
303  tmp_offset = child.getOffset() + static_cast<uint32_t>(child.getSize());
304  }
305  // STD140: Structures are padded to their alignment so that a[n] ==> sizeof(a[0]) == sizeof[a] / n
306  tmp_offset = align(tmp_offset, _baseAlignment);
307  tmp_offset = align(tmp_offset, _minDynamicAlignment);
308  _arrayMemberSize = tmp_offset;

311  _size = tmp_offset;
312  }

619  : _mappedMemory(nullptr), _level(level), _prototype(entry)
620  {
621  _indices[0] = elementArrayIndex;

628  void init(uint32_t dynamicSlice = 0)
629  {
630  // Get the offset INSIDE CURRENT LEVEL.
631  _offset = _prototype.getArrayElementOffset(_indices[0]);
632  const StructuredMemoryEntry* parent = _prototype.getParent();
633  size_t level = 1; // How many levels up we have gone
634  uint64_t dynamicSliceSize = 0;
635  uint32_t mappedDynamicSlice = 0;
636  while (parent) // Until we reach the root element
637  {
638  debug_assertion(parent->getNumArrayElements() > _indices[level], "StructuredBufferViewElement: Attempted out-of-bounds access in getOffset");
639  _offset += parent->getArrayElementOffset(_indices[level++]);
640  dynamicSliceSize = parent->getSize();
641  mappedDynamicSlice = parent->getMappedDynamicSlice();
642  // store the mapped memory so we only have to do this lookup once rather than each time setValue is called
643  _mappedMemory = parent->getMappedMemory();
644  parent = parent->getParent();
645  }
646 
647  // at this point dynamicSliceSize matches the root size
648  debug_assertion(dynamicSlice >= mappedDynamicSlice, "StructuredBufferViewElement: Mapped dynamic slice must be greater than or equal to the current dynamic slice");
649  uint32_t sliceOffset = (dynamicSlice * static_cast<uint32_t>(dynamicSliceSize)) - (mappedDynamicSlice * static_cast<uint32_t>(dynamicSliceSize));
650  _offset += sliceOffset;
651  }

962  }

972  }
