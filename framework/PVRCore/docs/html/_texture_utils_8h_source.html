16 inline glm::vec2 hammersley(uint32_t i, uint32_t N)
17 {
18  // Radical inverse based on http://holger.dammertz.org/stuff/notes_HammersleyOnHemisphere.html
19  uint32_t bits = (i << 16u) | (i >> 16u);
20  bits = ((bits & 0x55555555u) << 1u) | ((bits & 0xAAAAAAAAu) >> 1u);
21  bits = ((bits & 0x33333333u) << 2u) | ((bits & 0xCCCCCCCCu) >> 2u);
22  bits = ((bits & 0x0F0F0F0Fu) << 4u) | ((bits & 0xF0F0F0F0u) >> 4u);
23  bits = ((bits & 0x00FF00FFu) << 8u) | ((bits & 0xFF00FF00u) >> 8u);
24  float rdi = float(bits) * static_cast<float>(2.3283064365386963e-10);
25  return glm::vec2(float(i) / float(N), rdi);
26 }

40 // Sample a half-vector in world space
41 // Based on http://blog.selfshadow.com/publications/s2013-shading-course/karis/s2013_pbs_epic_slides.pdf
42 glm::vec3 importanceSampleGGX(glm::vec2 Xi, float roughness, glm::vec3 N)
43 {
44  // Maps a 2D point to a hemisphere with spread based on roughness
45  float a = roughness * roughness;
46  float phi = 2.0f * glm::pi<float>() * Xi.x;
47  float cosTheta = sqrt(glm::clamp((1.0f - Xi.y) / (1.0f + (a * a - 1.0f) * Xi.y), 0.0f, 1.0f));
48  float sinTheta = sqrt(glm::clamp(1.0f - cosTheta * cosTheta, 0.0f, 1.0f));
49 
50  glm::vec3 H = glm::vec3(sinTheta * cos(phi), sinTheta * sin(phi), cosTheta);
51 
52  glm::vec3 up = glm::abs(N.z) < 0.999 ? glm::vec3(0, 0, 1) : glm::vec3(1, 0, 0);
53  glm::vec3 tangent = glm::normalize(glm::cross(up, N));
54  glm::vec3 bitangent = glm::cross(N, tangent);
55 
56  return glm::normalize(tangent * H.x + bitangent * H.y + N * H.z);
57 }

59 // http://blog.selfshadow.com/publications/s2013-shading-course/karis/s2013_pbs_epic_notes_v2.pdf
60 glm::vec2 integrateBRDF(float roughness, float NoV)
61 {
62  const glm::vec3 N = glm::vec3(0.0, 0.0, 1.0); // normal always pointing forward.
63  const glm::vec3 V = glm::vec3(sqrt(glm::clamp(1.0 - NoV * NoV, 0.0, 1.0)), 0.0, NoV);
64  float A = 0.0f;
65  float B = 0.0f;
66 
67  uint32_t numSamples = 1024u;
68  for (uint32_t i = 0u; i < numSamples; ++i)
69  {
70  glm::vec2 Xi = hammersley(i, numSamples);
71  glm::vec3 H = importanceSampleGGX(Xi, roughness, N);
72  glm::vec3 L = 2.0f * glm::dot(V, H) * H - V;
73 
74  float NoL = glm::max(glm::dot(N, L), 0.0f);
75  if (NoL > 0.0f)
76  {
77  float NoH = glm::max(glm::dot(N, H), 0.001f);
78  float VoH = glm::max(glm::dot(V, H), 0.001f);
79  float NoV = glm::max(glm::dot(N, V), 0.001f);
80 
81  const float G = gSmith(NoL, NoV, roughness);
82 
83  const float G_Vis = (G * VoH) / (NoH * NoV /*avoid division by zero*/);
84  const float Fc = pow(1.0f - VoH, 5.0f);
85 
86  A += (1.0f - Fc) * G_Vis;
87  B += Fc * G_Vis;
88  }
89  }
90 
91  return glm::vec2(A, B) / float(numSamples);
92 }
