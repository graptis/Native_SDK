67 inline bool isDebuggerPresent()
68 {
69  // only check once for whether the debugger is present as this may not be efficient to determine
70  static bool isUsingDebugger = false;
71  static bool haveCheckedForDebugger = false;
72  if (!haveCheckedForDebugger)
73  {
74 #if defined(_MSC_VER)
75  if (IsDebuggerPresent())
76  {
77  isUsingDebugger = true;
78  }
79 #elif defined(__linux__)
80  // reference implementation taken from: https://stackoverflow.com/a/24969863
81  char buf[1024];
82 
83  int status_fd = open("/proc/self/status", O_RDONLY);
84  if (status_fd == -1)
85  {
86  isUsingDebugger = false;
87  }
88  else
89  {
90  ssize_t num_read = read(status_fd, buf, sizeof(buf) - 1);
91  if (num_read > 0)
92  {
93  static const char TracerPid[] = "TracerPid:";
94  char* tracer_pid;
95 
96  buf[num_read] = 0;
97  tracer_pid = strstr(buf, TracerPid);
98  if (tracer_pid)
99  {
100  isUsingDebugger = !!atoi(tracer_pid + sizeof(TracerPid) - 1);
101  }
102  }
103  }
104 #endif
105  haveCheckedForDebugger = true;
106  }
107 
108  return isUsingDebugger;
