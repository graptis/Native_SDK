441  void compact()
442  {
443  // We can do that because the last remove() tears down all datastructures used.
444  if (!myindex.size())
445  {
446  return;
447  }
448  // First, make sure there is something to compact...
449  deleteditemlisttype_::iterator unused_spot = myDeletedItems.begin();
450  while (myDeletedItems.size() && unused_spot != myDeletedItems.end())
451  {
452  // Last item in the storage vector.
453  size_t last = mystorage.size();
454 
455  // 1) Trim the end of the vector...
456  while (last-- && mystorage[last].isUnused)
457  {
458  mystorage.pop_back();
459  // Rinse, repeat. If size is zero, there is nothing to do.
460  }
461 
462  // Either the storage is empty
463  if (mystorage.empty())
464  {
465  myDeletedItems.clear();
466  // the rest should already be cleared...
467  // the loop will exit naturally
468  }
469  else // Or we can have find its last item!
470  {
471  // 2)Trim any items from the end of the unused spots list that may have been trimmed off by the vector...
472  while (unused_spot != myDeletedItems.end() && *unused_spot >= last)
473  {
474  myDeletedItems.erase(unused_spot++);
475  }
476  // Any spots left?
477  if (unused_spot != myDeletedItems.end())
478  {
479  // Do the actual data movement. After all we've been through, we know that
480  // i. The last item of the vector is a valid item (guaranteed by 1)
481  // ii. The unused spot is not out of bounds of the vector
482  // iii.
483  // Copy by hand(as we have not defined a move assignment operator for compatibility reasons.
484  // Also, since the std::string does not throw, this makes easier to reason about exceptions.
485 
486  mystorage[*unused_spot].value = std::move(mystorage[last].value);
487  mystorage[*unused_spot].key = std::move(mystorage[last].key);
488 
489  mystorage[*unused_spot].isUnused = false;
490  myindex[mystorage[*unused_spot].key] = *unused_spot;
491  mystorage.pop_back();
492  myDeletedItems.erase(unused_spot++);
493  } // else : No action needed - unused spots has been trimmed off completely, so no movement is possible, or necessary...
494  }
495  }

539  bool relocate(const IndexType_& key, size_t index)
540  {
541  typename maptype_::iterator found = myindex.find(key);
542  if (found == myindex.end())
543  {
544  return false;
545  }
546  size_t old_index = myindex[key];
547  if (index == old_index)
548  {
549  return true;
550  } // No-op
551  if (index + 1 > mystorage.size()) // Storage not big enough.
552  {
553  // Grow, and mark unused all required items. Need to add all spots (But the last) to unusedspots.
554  size_t oldsize = mystorage.size();
555  mystorage.resize(index + 1);
556  for (size_t i = oldsize; i < index; ++i)
557  {
558  myDeletedItems.push_front(i);
559  mystorage[i].isUnused = 1;
560  }
561  mystorage.back() = mystorage[old_index];
562  removefromvector(old_index);
563  }
564  else if (mystorage[index].isUnused) // Lucky! Storage is big enough, and the item is not used!
565  {
566  deleteditemlisttype_::iterator place = std::find(myDeletedItems.begin(), myDeletedItems.end(), index);
567  assert(place != myDeletedItems.end()); // Shouldn't happen! Ever!
568  myDeletedItems.erase(place);
569  mystorage[index] = mystorage[old_index];
570  removefromvector(old_index);
571  }
572  else // Whoops! Space is already occupied. Swap with the old item!
573  {
574  myindex[mystorage[index].key] = old_index;
575  std::swap(mystorage[index], mystorage[old_index]);
576  }
577  myindex[key] = index;
578  return true;
579  }

604  {
605  if (index == (mystorage.size() - 1))
