35 inline static bool isDebuggerPresent()
36 {
37  // only check once for whether the debugger is present as this may not be efficient to determine
38  static bool isUsingDebugger = false;
39  static bool haveCheckedForDebugger = false;
40  if (!haveCheckedForDebugger)
41  {
42 #if defined(_MSC_VER)
43  if (IsDebuggerPresent())
44  {
45  isUsingDebugger = true;
46  }
47 #elif defined(__linux__)
48  // reference implementation taken from: https://stackoverflow.com/a/24969863
49  char buf[1024];
50 
51  int status_fd = open("/proc/self/status", O_RDONLY);
52  if (status_fd == -1)
53  {
54  isUsingDebugger = false;
55  }
56  else
57  {
58  ssize_t num_read = read(status_fd, buf, sizeof(buf) - 1);
59  if (num_read > 0)
60  {
61  static const char TracerPid[] = "TracerPid:";
62  char* tracer_pid;
63 
64  buf[num_read] = 0;
65  tracer_pid = strstr(buf, TracerPid);
66  if (tracer_pid)
67  {
68  isUsingDebugger = !!atoi(tracer_pid + sizeof(TracerPid) - 1);
69  }
70  }
71  }
72 #endif
73  haveCheckedForDebugger = true;
74  }
75 
76  return isUsingDebugger;
77 }
