464  void parseCmdLine(const char* const cmdLine)
465  {
466  size_t len;
467  int nIn, nOut;
468  bool bInQuotes;
469  ParsedCommandLine::Option opt;
470 
471  if (!cmdLine)
472  {
473  return;
474  }
475 
476  // Take a copy of the original
477  len = strlen(cmdLine) + 1;
478 
479  // Take a copy to be edited
480  _data.resize(len);
481 
482  // Break the command line into options
483  bInQuotes = false;
484  opt.arg = NULL;
485  opt.val = NULL;
486  nIn = -1;
487  nOut = 0;
488  do
489  {
490  ++nIn;
491  if (cmdLine[nIn] == '"')
492  {
493  bInQuotes = !bInQuotes;
494  }
495  else
496  {
497  if (bInQuotes && cmdLine[nIn] != 0)
498  {
499  if (!opt.arg)
500  {
501  opt.arg = &_data[nOut];
502  }
503 
504  _data[nOut++] = cmdLine[nIn];
505  }
506  else
507  {
508  switch (cmdLine[nIn])
509  {
510  case '=':
511  _data[nOut++] = 0;
512  opt.val = &_data[nOut];
513  break;
514 
515  case ' ':
516  case '\t':
517  case '\0':
518  _data[nOut++] = 0;
519  if (opt.arg || opt.val)
520  {
521  // Add option to list
522  _commandLine._options.push_back(opt);
523 
524  opt.arg = NULL;
525  opt.val = NULL;
526  }
527  break;
528 
529  default:
530  if (!opt.arg)
531  {
532  opt.arg = &_data[nOut];
533  }
534 
535  _data[nOut++] = cmdLine[nIn];
536  break;
537  }
538  }
539  }
540  } while (cmdLine[nIn]);
