304 inline uint64_t generatePascalTriangleRow(const uint32_t row, std::vector<uint64_t>& pascalCoefficients)
305 {
306  // Each entry of any given row of the Pascal Triangle is constructed by adding the number above and to the left with the number above and to the right.
307  // Entries which fall outside of the Pascal Triangle are treated as having a 0 value. The first row consists of a single entry with a value of 1.
308  // The following shows the first 4 rows of the Pascal Triangle
309  // 1 row 0 ... sum = 1
310  // 1 1 row 1 ... sum = 2
311  // 1 2 1 row 2 ... sum = 4
312  // 1 3 3 1 row 3 ... sum = 8
313 
314  pascalCoefficients.push_back(1);
315  uint64_t sum = pascalCoefficients.back();
316  for (uint32_t i = 0; i < row; i++)
317  {
318  uint64_t val = pascalCoefficients[i] * (row - i) / (i + 1);
319  pascalCoefficients.push_back(val);
320  sum += pascalCoefficients.back();
321  }
322 
323  return sum;
324 }

331 inline void adjustOffsetsAndWeightsForLinearSampling(const uint32_t halfCoefficientsMinusOne, std::vector<double>& weights, std::vector<double>& offsets)
332 {
333  std::vector<double> adjustedWeights;
334  std::vector<double> adjustedOffsets;
335 
336  // if kernel size minus 1 is divisible by 2 then we have a central sample with offset 0
337  if (halfCoefficientsMinusOne % 2 == 0)
338  {
339  uint32_t i = 0;
340  for (; i < halfCoefficientsMinusOne - 1; i += 2)
341  {
342  adjustedWeights.push_back(weights[i] + weights[i + 1]);
343  double adjustedOffset = ((offsets[i] * weights[i]) + (offsets[i + 1] * weights[i + 1])) / adjustedWeights.back();
344  adjustedOffsets.push_back(adjustedOffset);
345  }
346 
347  adjustedWeights.push_back(weights[halfCoefficientsMinusOne]);
348  adjustedOffsets.push_back(0.0f);
349 
350  for (i = halfCoefficientsMinusOne + 1; i < offsets.size(); i += 2)
351  {
352  adjustedWeights.push_back(weights[i] + weights[i + 1]);
353  double adjustedOffset = ((offsets[i] * weights[i]) + (offsets[i + 1] * weights[i + 1])) / adjustedWeights.back();
354  adjustedOffsets.push_back(adjustedOffset);
355  }
356  }
357  else // otherwise we have to duplicate the central sample *but* this means we can handle 3x3 using 2x2 samples
358  {
359  uint32_t i = 0;
360  for (; i < halfCoefficientsMinusOne; i += 2)
361  {
362  double adjustedOffset = 0.0;
363  if (i == halfCoefficientsMinusOne - 1)
364  {
365  adjustedWeights.push_back(weights[i] + weights[i + 1] * 0.5);
366  adjustedOffset = ((offsets[i] * weights[i]) + (offsets[i + 1] * weights[i + 1] * 0.5)) / adjustedWeights.back();
367  }
368  else
369  {
370  adjustedWeights.push_back(weights[i] + weights[i + 1]);
371  adjustedOffset = ((offsets[i] * weights[i]) + (offsets[i + 1] * weights[i + 1])) / adjustedWeights.back();
372  }
373  adjustedOffsets.push_back(adjustedOffset);
374  }
375 
376  for (i = halfCoefficientsMinusOne; i < offsets.size(); i += 2)
377  {
378  double adjustedOffset = 0.0;
379  if (i == halfCoefficientsMinusOne)
380  {
381  adjustedWeights.push_back(weights[i] * 0.5 + weights[i + 1]);
382  adjustedOffset = ((offsets[i] * weights[i] * 0.5) + (offsets[i + 1] * weights[i + 1])) / adjustedWeights.back();
383  }
384  else
385  {
386  adjustedWeights.push_back(weights[i] + weights[i + 1]);
387  adjustedOffset = ((offsets[i] * weights[i]) + (offsets[i + 1] * weights[i + 1])) / adjustedWeights.back();
388  }
389  adjustedOffsets.push_back(adjustedOffset);
390  }
391  }

394  weights.clear();
395 
396  for (uint32_t i = 0; i < adjustedWeights.size(); i++)

421 inline void generateGaussianKernelWeightsAndOffsets(uint32_t kernelSize, bool truncateCoefficients, bool useLinearSamplerOptimization, std::vector<double>& weights,
422  std::vector<double>& offsets, float minimumAcceptableCoefficient = 0.0001f)
423 {
424  // The specified kernel size must be odd
425  // Note that it would be possible to extend this utility function to support even sized kernel sizes or even fractional kernel sizes
426  assert((kernelSize % 2) == 1);
427 
428  // the starting row of the Pascal Triangle being used
429  uint32_t pascalRow = kernelSize - 1;
430 
431  // The number of coefficients minus 1 halved
432  // This variable is used to get the index of the coefficient used for an offset of 0
433  // (numCoefficients - 1) / 2
434  uint32_t halfCoefficientsMinusOne = pascalRow / 2;
435 
436  // stores the set of pascal coefficients
437  std::vector<uint64_t> pascalCoefficients;
438 
439  // the number of coefficients skipped due to minimum coefficient checks
440  uint32_t numCoefficientsSkipped = 0;
441 
442  // Ignoring negligible coefficients - we'll now attempt to find a row which provides enough coefficients for the given kernel size whilst not falling below the
443  // given minimal coefficient value.
444  if (truncateCoefficients)
445  {
446  uint32_t currentRow = pascalRow;
447  bool foundSuitableRow = false;
448 
449  // only accept rows where we have kernelSize coefficients larger than the min coefficient
450  while (!foundSuitableRow)
451  {
452  // clear the offsets and weights
453  offsets.clear();
454  weights.clear();
455 
456  // clear the set of pascal triangle coefficients
457  pascalCoefficients.clear();
458 
459  // get the pascal coefficients for the current row of the triangle
460  uint64_t pascalSum = generatePascalTriangleRow(currentRow, pascalCoefficients);
461 
462  halfCoefficientsMinusOne = currentRow / 2;
463 
464  // check how many of the coefficients are negligible in size and therefore should be ignored
465  numCoefficientsSkipped = 0;
466  for (uint32_t i = halfCoefficientsMinusOne; i < pascalCoefficients.size(); i++)
467  {
468  double currentWeight = static_cast<double>(pascalCoefficients[i]) / pascalSum;
469  if (currentWeight < minimumAcceptableCoefficient)
470  {
471  numCoefficientsSkipped++;
472  }
473  }
474 
475  // if there aren't enough coefficients left to fulfill the requirements for the kernel size then continue to the next row
476  if ((halfCoefficientsMinusOne + 1) - numCoefficientsSkipped < (pascalRow / 2 + 1))
477  {
478  currentRow += 2;
479  foundSuitableRow = false;
480  continue;
481  }
482 
483  // if negligible coefficients are to be removed we must also update the overall sums used to give them weighting
484  // otherwise repeated blurring will result in darkening of the image
485  uint64_t adjustedPascalSum = pascalSum;
486  // the extra coefficients which aren't needed which do match the conditions for non-negligibility but would result in us taking extra coefficients
487  uint64_t unrequiredCoefficients = ((pascalCoefficients.size() - kernelSize - (numCoefficientsSkipped * 2)) / 2);
488  for (uint32_t i = 0; i < numCoefficientsSkipped + unrequiredCoefficients; i++)
489  {
490  adjustedPascalSum -= 2 * pascalCoefficients[pascalCoefficients.size() - 1 - i];
491  }
492 
493  // add the non-negligible coefficients to the weights and offsets buffers
494  uint32_t numCoefficients = static_cast<uint32_t>(pascalCoefficients.size() - (2 * (numCoefficientsSkipped + unrequiredCoefficients)));
495  addCoefficients(adjustedPascalSum, halfCoefficientsMinusOne, numCoefficients, pascalCoefficients, weights, offsets);
496  halfCoefficientsMinusOne = static_cast<uint32_t>((offsets.size() - 1) / 2);
497  foundSuitableRow = true;
498  }
499  }
500  else
501  {
502  // Generate the Pascal Triangle row and
503  uint64_t pascalSum = generatePascalTriangleRow(pascalRow, pascalCoefficients);
504  // Store the Pascal Triangle coefficients for the given row
505  addCoefficients(pascalSum, halfCoefficientsMinusOne, static_cast<uint32_t>(pascalCoefficients.size()), pascalCoefficients, weights, offsets);
506  }
507 
508  // If using the Linear Sampling optimisation then adjust the Gaussian weights and offsets accordingly
509  if (useLinearSamplerOptimization)
510  {
511  adjustOffsetsAndWeightsForLinearSampling(halfCoefficientsMinusOne, weights, offsets);
512  }
513 }

Generates a set of Gaussian weights and offsets based on the given configuration values. This function makes use of the Pascal Triangle for calculating the Gaussian distribution. The Gaussian function is a distribution function of the normal distribution who's discrete equivalent is the binomial distribution for which the Pascal Triangle models. The Pascal Triangle provides us with a convenient and efficient mechanism for calculating the Gaussian weights and offsets required. Our method of generating Gaussian weights and offsets was inspired by http://rastergrid.com/blog/2010/09/efficient-gaussian-blur-with-linear-sampling/ .
