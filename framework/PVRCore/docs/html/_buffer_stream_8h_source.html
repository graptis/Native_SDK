137  virtual void seek(long offset, SeekOrigin origin) const
138  {
139  long newOffset = 0;
140 
141  if (!_currentPointer || !_originalData)
142  {
143  if (offset)
144  {
145  throw FileIOError("BufferStream::seek: Attempt to seek from empty stream");
146  }
147  }
148  else
149  {
150  switch (origin)
151  {
152  case Stream::SeekOriginFromStart:
153  {
154  newOffset = static_cast<long>(CLAMP(static_cast<int64_t>(offset), 0, static_cast<int64_t>(_bufferSize)));
155 
156  _bufferPosition = newOffset;
157  _currentPointer = const_cast<void*>(static_cast<const void*>(static_cast<const unsigned char*>(_originalData) + _bufferPosition));
158  break;
159  }
160  case Stream::SeekOriginFromCurrent:
161  {
162  int64_t maxOffset = _bufferSize - _bufferPosition;
163  int64_t minOffset = -1 * static_cast<int64_t>(_bufferPosition);
164 
165  newOffset = static_cast<long>(CLAMP(offset, static_cast<long>(minOffset), static_cast<long>(maxOffset)));
166 
167  _bufferPosition += newOffset;
168  _currentPointer = static_cast<void*>(static_cast<char*>(_currentPointer) + newOffset);
169  break;
170  }
171  case Stream::SeekOriginFromEnd:
172  {
173  newOffset = static_cast<long>(CLAMP(offset, static_cast<long>(-1 * (static_cast<int64_t>(_bufferPosition))), 0));
174

177  break;
