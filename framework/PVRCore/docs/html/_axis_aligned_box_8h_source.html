79 inline void getFrustumPlanes(pvr::Api api, const glm::mat4& projection_from_world, ViewingFrustum& frustum_out)
80 {
81  const glm::vec4 row0 = glm::row(projection_from_world, 0);
82  const glm::vec4 row1 = glm::row(projection_from_world, 1) * (api == pvr::Api::Vulkan ? -1.f : 1.f);
83  const glm::vec4 row2 = glm::row(projection_from_world, 2);
84  const glm::vec4 row3 = glm::row(projection_from_world, 3);
85 
86  frustum_out.minusX = row3 + row0;
87  frustum_out.plusX = row3 - row0;
88  frustum_out.minusY = row3 + row1;
89  frustum_out.plusY = row3 - row1;
90  frustum_out.minusZ = row3 + row2;
91  frustum_out.plusZ = row3 - row2;
92 
93  frustum_out.minusX = frustum_out.minusX * (1 / glm::length(glm::vec3(frustum_out.minusX)));
94  frustum_out.plusX = frustum_out.plusX * (1 / glm::length(glm::vec3(frustum_out.plusX)));
95  frustum_out.minusY = frustum_out.minusY * (1 / glm::length(glm::vec3(frustum_out.minusY)));
96  frustum_out.plusY = frustum_out.plusY * (1 / glm::length(glm::vec3(frustum_out.plusY)));
