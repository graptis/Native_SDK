104 inline glm::vec3 IntersectPlanes(glm::vec4 P0, glm::vec4 P1, glm::vec4 P2)
105 {
106  glm::vec3 bxc = glm::cross(glm::vec3(P1), glm::vec3(P2));
107  glm::vec3 cxa = glm::cross(glm::vec3(P2), glm::vec3(P0));
108  glm::vec3 axb = glm::cross(glm::vec3(P0), glm::vec3(P1));
109  glm::vec3 r = -P0.w * bxc - P1.w * cxa - P2.w * axb;
110  return r * (1 / glm::dot(glm::vec3(P0), bxc));
111 }
